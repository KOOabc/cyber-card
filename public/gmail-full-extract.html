自动提取所有用户完整信息（姓名、公司、手机、邮箱等）

📌 步骤1：添加书签工具
⚠️ 重要：请拖动（不是点击）下方按钮到浏览器书签栏！
如果书签栏不可见，按 Ctrl+Shift+B (Windows) 或 Cmd+Shift+B (Mac)
🎯 提取完整用户数据
💡 拖动方法：
1. 鼠标按住上方按钮
2. 拖动到浏览器顶部书签栏
3. 松开鼠标
4. 完成！
🚀 步骤2：使用工具
  1. 打开 Gmail
  2. 在搜索框输入：赛博名片提交 或 from:service@emailjs.com
  3. 等待邮件列表加载完成
  4. 点击书签栏中的 "提取完整用户数据" 按钮
  5. 工具会自动：
    • ✅ 逐个打开邮件
    • ✅ 提取完整的用户信息
    • ✅ 包括：姓名、公司、职位、手机、微信、邮箱等
    • ✅ 自动保存为 CSV 文件
  6. 等待提取完成（每封邮件约2-3秒）
  7. 自动下载 CSV 文件，用 Excel 打开查看
⚠️ 注意事项：
• 提取过程中不要关闭或切换 Gmail 页面
• 默认最多提取 50 封邮件（可在代码中修改）
• 如果邮件很多，建议分批提取

🔧 高级：Chrome 控制台版本

如果拖动书签不方便，可以使用控制台版本：

  1. 在 Gmail 页面按 F12 打开开发者工具
  2. 切换到 Console 标签
  3. 复制下方脚本并粘贴运行
// Gmail 完整数据提取脚本 - 赛博名片
(async function() {
  alert('🚀 开始提取完整数据...\n\n这可能需要几秒到几分钟。\n请不要关闭页面！');
  
  let allData = [];
  let processed = 0;
  
  const wait = (ms) => new Promise(r => setTimeout(r, ms));
  
  // 提取单封邮件的完整数据
  async function extractFromEmail() {
    const emailBody = document.querySelector('[role="main"] [data-message-id]');
    if (!emailBody) return null;
    
    const text = emailBody.innerText;
    const data = {};
    
    // 通用提取函数
    const extract = (label) => {
      const regex = new RegExp(label + '[\\s\\u3000]*(.+?)(?:\\n|$)', 'i');
      const match = text.match(regex);
      return match ? match[1].trim() : '';
    };
    
    // 提取所有字段
    data.submitTime = extract('提交时间');
    data.realName = extract('姓名');
    data.company = extract('公司');
    data.title = extract('职位');
    data.phone = extract('手机号?');
    data.wechat = extract('微信');
    data.email = extract('邮箱');
    data.aiDirection = extract('AI\\s*方向');
    data.isOversea = extract('出海计划');
    data.overseaMarkets = extract('目标市场');
    data.cloudNeed = extract('算力需求');
    data.businessIntro = extract('业务介绍');
    data.displayName = extract('展示名称');
    data.titles = extract('职业身份');
    data.organization = extract('所属机构');
    data.bio = extract('个人简介');
    data.skills = extract('技能标签');
    data.templateId = extract('选择模板');
    
    // 验证数据有效性
    const validFields = Object.values(data).filter(v => v && v !== '无' && v !== '🈚').length;
    return validFields >= 5 ? data : null;
  }
  
  // 处理所有邮件
  async function processEmails() {
    // 确保在列表视图
    const backBtn = document.querySelector('[aria-label*="返回"], [aria-label*="Back"]');
    if (backBtn) {
      backBtn.click();
      await wait(800);
    }
    
    // 获取所有邮件行
    const threads = document.querySelectorAll('[role="main"] [role="row"]');
    const total = threads.length;
    
    if (total === 0) {
      alert('❌ 未找到邮件，请确保在 Gmail 邮件列表页面');
      return;
    }
    
    console.log(`📧 找到 ${total} 封邮件，开始提取...`);
    
    // 限制最多处理 50 封（避免太慢）
    for (let i = 0; i < Math.min(threads.length, 50); i++) {
      const thread = threads[i];
      const subject = thread.querySelector('[data-legacy-thread-id]')?.textContent || '';
      
      // 只处理赛博名片相关邮件
      if (!subject.includes('赛博名片') && !subject.includes('新的提交')) {
        continue;
      }
      
      console.log(`📩 处理第 ${i + 1} 封: ${subject}`);
      
      // 点击打开邮件
      thread.click();
      await wait(2000); // 等待邮件加载
      
      // 提取数据
      const data = await extractFromEmail();
      if (data) {
        allData.push(data);
        processed++;
        console.log(`✅ 已提取 ${processed} 条数据`);
      }
      
      // 返回列表
      const back = document.querySelector('[aria-label*="返回"], [aria-label*="Back"]');
      if (back) {
        back.click();
        await wait(1000);
      }
    }
    
    if (allData.length === 0) {
      alert('❌ 未提取到有效数据，请检查邮件格式');
      return;
    }
    
    // 导出为 CSV
    const headers = [
      '序号', '提交时间', '姓名', '公司', '职位', '手机', '微信', '邮箱',
      'AI方向', '出海计划', '目标市场', '算力需求', '业务介绍',
      '展示名称', '职业身份', '所属机构', '个人简介', '技能标签', '模板'
    ];
    
    const rows = allData.map((d, i) => [
      i + 1,
      d.submitTime,
      d.realName,
      d.company,
      d.title,
      d.phone,
      d.wechat,
      d.email,
      d.aiDirection,
      d.isOversea,
      d.overseaMarkets,
      d.cloudNeed,
      d.businessIntro,
      d.displayName,
      d.titles,
      d.organization,
      d.bio,
      d.skills,
      d.templateId
    ]);
    
    const csv = [headers, ...rows]
      .map(row => row.map(cell => `"${String(cell || '').replace(/"/g, '""')}"`).join(','))
      .join('\n');
    
    // 下载 CSV
    const blob = new Blob(['\ufeff' + csv], { type: 'text/csv;charset=utf-8;' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `赛博名片完整数据_${new Date().toISOString().slice(0, 10)}.csv`;
    a.click();
    
    console.log('✅ 提取完成！');
    console.table(allData);
    alert(`✅ 提取完成！\n\n成功提取 ${allData.length} 条完整数据\n已下载为 CSV 文件，可用 Excel 打开`);
  }
  
  processEmails();
})();
