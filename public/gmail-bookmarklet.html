// Gmail 批量提取脚本
(async function() {
  let data = [];
  
  // 滚动加载更多邮件
  const container = document.querySelector('[role="main"]');
  if (!container) {
    alert('❌ 请确保在 Gmail 邮件列表页面');
    return;
  }
  
  console.log('🔄 开始加载邮件...');
  let lastHeight = 0;
  for (let i = 0; i < 10; i++) {
    container.scrollTo(0, container.scrollHeight);
    await new Promise(r => setTimeout(r, 1000));
    if (container.scrollHeight === lastHeight) break;
    lastHeight = container.scrollHeight;
  }
  
  // 提取邮件数据
  const rows = document.querySelectorAll('[role="main"] [role="row"]');
  console.log(`📧 找到 ${rows.length} 封邮件`);
  
  rows.forEach((row, index) => {
    const subject = row.querySelector('[data-legacy-thread-id]')?.textContent.trim() || '';
    if (subject.includes('赛博名片') || subject.includes('新的提交')) {
      const sender = row.querySelector('[email]')?.getAttribute('email') || '';
      const preview = row.querySelector('.y2')?.textContent.trim() || '';
      data.push({ 序号: index + 1, 主题: subject, 发件人: sender, 预览: preview });
    }
  });
  
  console.log(`✅ 提取到 ${data.length} 条数据`);
  console.table(data);
  
  // 下载为 CSV
  const headers = ['序号', '主题', '发件人', '预览'];
  const csv = [
    headers,
    ...data.map(d => [d.序号, d.主题, d.发件人, d.预览])
  ].map(row => row.map(cell => `"${String(cell).replace(/"/g, '""')}"`).join(',')).join('\n');
  
  const blob = new Blob(['\ufeff' + csv], { type: 'text/csv;charset=utf-8;' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `Gmail数据_${new Date().toISOString().slice(0, 10)}.csv`;
  a.click();
  
  alert(`✅ 已提取 ${data.length} 封邮件\n已下载为 CSV 文件`);
})();
⚠️ 重要提示：
• 这些工具仅提取邮件列表的预览信息，完整数据需要打开邮件查看
• 建议使用 邮件解析工具 来提取完整的用户数据
• 所有操作均在本地浏览器完成，不会上传数据到服务器
